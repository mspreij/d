# Specify roles in playbook:
- name: random playbook
  roles:
    - \e[32;1mfoo\e[0m           # regular role names
    - \e[32;1mbar\e[0m
    - role: \e[32;1mfoobar\e[0m  # with this format you can add additional parameters:
      vars:
        var_specific_to_foobar: 123
      when: some_boolean

# run only a specific task, which Does Not Need vars, defaults or dependencies from its Role.
- name: do some shit
  become: true
  gather_facts: true
  hosts:
    - mercury
  tasks:
    - include_tasks: roles/container-webhost/tasks/firewall.yml

# run a specific task that DOES need stuff from the role:
- name: do some shit
  become: true
  gather_facts: true
  hosts:
    - mercury
  tasks:
    - include_role:
        name: container-webhost
        tasks_from: firewall.yml
